def pore_volume(cube_length_nm, void_fraction):
    """
    Simple cubic model.
    cube_length_nm: cell edge length (nm)
    void_fraction: porosity (void fraction)
    Returns pore volume (nm^3)
    """
    v_cell = cube_length_nm**3
    return v_cell * void_fraction
